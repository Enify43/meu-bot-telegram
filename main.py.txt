import os
import logging
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, ContextTypes, filters

logging.basicConfig(level=logging.INFO)

TOKEN = os.getenv("BOT_TOKEN")

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("Oi! Me envie um link que eu respondo ðŸ§©")

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.strip()
    # Aqui vocÃª vai colocar a lÃ³gica de converter link + formatar mensagem depois
    await update.message.reply_text(f"Recebi: {text}\n\n(Em breve vou converter e agendar ðŸ˜‰)")

def main():
    if not TOKEN:
        raise RuntimeError("Faltou a variÃ¡vel de ambiente BOT_TOKEN")

    app = Application.builder().token(TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    # Long polling (mais simples para comeÃ§ar)
    app.run_polling(allowed_updates=Update.ALL_TYPES)

if __name__ == "__main__":
    main()
